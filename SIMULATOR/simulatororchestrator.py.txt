"""
Latency-Aware SFC Orchestration Algorithm
Implements the two-phase heuristic from Section III.C of the manuscript.

Algorithm Overview:
  Phase 1: Latency-aware VNF placement (greedy ranking with resource awareness)
  Phase 2: Security-aware path routing (modified Dijkstra with policy constraints)
"""

import numpy as np
import networkx as nx
from typing import List, Dict, Tuple, Optional
import logging

logger = logging.getLogger(__name__)

class LatencyAwareOrchestrator:
    """Two-phase heuristic for latency-aware SFC orchestration."""
    
    def __init__(self, network: nx.Graph, alpha: float = 0.7, beta: float = 0.8):
        """
        Initialize orchestrator.
        
        Args:
            network: NetworkX graph representing physical infrastructure
            alpha: Weight for latency vs. security in objective function (0-1)
                   Higher alpha = more weight on latency minimization
            beta: Weight for latency vs. security in routing weights (0-1)
                  Higher beta = more weight on latency in path selection
        """
        self.G = network
        self.alpha = alpha
        self.beta = beta
        self.resource_usage = {
            node: {'cpu': 0, 'memory': 0} 
            for node in network.nodes()
        }
        logger.info(f"Initialized orchestrator: alpha={alpha}, beta={beta}")
    
    def _compute_node_score(self, v: str, source: str, vnf: Dict, 
                          available_resources: Dict) -> float:
        """
        Compute placement score for node v (Eq. 2 in manuscript).
        
        Score(v) = [1 / (d_prop + d_proc)] × (C_avail / C_total)
        
        Args:
            v: Candidate node ID
            source: Source node of SFC request
            vnf: VNF specification dictionary
            available_resources: Dict of available CPU per node
            
        Returns:
            float: Higher score = better placement candidate
        """
        try:
            # Propagation delay from source to candidate node
            d_prop = nx.shortest_path_length(
                self.G, source, v, weight='latency'
            )
        except nx.NetworkXNoPath:
            # No path exists - assign very low score
            return -1e6
        
        # Processing delay for VNF on node (with small random variation)
        base_proc = vnf.get('processing_time', 1.0)
        node_speed = self.G.nodes[v].get('compute_speed', 1.0)
        d_proc = base_proc / node_speed * np.random.uniform(0.9, 1.1)
        
        # Resource availability ratio (favor nodes with more free capacity)
        cpu_avail = available_resources.get(v, 0)
        cpu_total = self.G.nodes[v].get('cpu_capacity', 1)
        resource_ratio = cpu_avail / cpu_total if cpu_total > 0 else 0
        
        # Combined score (Eq. 2)
        latency_term = 1.0 / (d_prop + d_proc + 1e-6)  # Avoid division by zero
        score = latency_term * resource_ratio
        
        return score
    
    def place_vnfs(self, request: Dict) -> Dict[str, str]:
        """
        Phase 1: Latency-aware VNF placement (greedy ranking).
        
        Places each VNF in chain order on the highest-scoring feasible node.
        
        Args:
            request: SFC request dictionary
            
        Returns:
            Dict mapping VNF ID -> node ID where placed
        """
        placement = {}
        
        # Track available resources (copy to avoid modifying global state)
        available = {
            node: {
                'cpu': self.G.nodes[node]['cpu_capacity'] - 
                      self.resource_usage[node]['cpu']
            }
            for node in self.G.nodes()
        }
        
        current_location = request['source']
        
        for vnf in request['chain']:
            # Find feasible nodes (sufficient CPU resources)
            feasible = [
                v for v in self.G.nodes() 
                if available[v]['cpu'] >= vnf['cpu_requirement']
            ]
            
            if not feasible:
                logger.warning(f"No feasible node for VNF {vnf['id']}")
                raise RuntimeError(f"Resource exhaustion: cannot place {vnf['id']}")
            
            # Score all feasible nodes
            scores = {
                v: self._compute_node_score(v, current_location, vnf, available)
                for v in feasible
            }
            
            # Select best node (highest score)
            best_node = max(scores, key=scores.get)
            
            # Place VNF and update resource tracking
            placement[vnf['id']] = best_node
            available[best_node]['cpu'] -= vnf['cpu_requirement']
            self.resource_usage[best_node]['cpu'] += vnf['cpu_requirement']
            
            # Update current location for next VNF placement
            current_location = best_node
            
            logger.debug(f"Placed {vnf['id']} on {best_node} (score={scores[best_node]:.3f})")
        
        return placement
    
    def _security_indicator(self, edge: Tuple[str, str], 
                          security_policy: str) -> float:
        """
        Compute security compliance indicator (Eq. 3 in manuscript).
        
        Returns 0 if edge satisfies policy, infinity otherwise.
        
        Security levels (simplified model):
          - 'none': No encryption/authentication
          - 'basic': TLS 1.2, basic auth
          - 'advanced': TLS 1.3, mutual auth, integrity checks
        """
        edge_security = self.G.edges[edge].get('security_level', 'none')
        
        # Policy requirements mapping
        policy_requirements = {
            'low': ['none', 'basic', 'advanced'],
            'medium': ['basic', 'advanced'],
            'high': ['advanced']
        }
        
        if edge_security in policy_requirements.get(security_policy, []):
            return 0.0
        return float('inf')
    
    def _compute_link_weight(self, u: str, v: str, security_policy: str) -> float:
        """
        Compute routing weight for link (u,v) (Eq. 3 in manuscript).
        
        w(e) = β × d_link(e) + (1-β) × I_sec(e)
        """
        # Base latency of link
        d_link = self.G.edges[u, v].get('latency', 1.0)
        
        # Security compliance penalty
        sec_penalty = self._security_indicator((u, v), security_policy)
        
        # Combined weight (Eq. 3)
        weight = self.beta * d_link + (1 - self.beta) * sec_penalty
        
        return weight
    
    def route_path(self, placement: Dict[str, str], request: Dict) -> List[str]:
        """
        Phase 2: Security-aware path routing (modified Dijkstra).
        
        Computes shortest path through placed VNFs with security constraints.
        
        Args:
            placement: Dict mapping VNF ID -> node ID
            request: SFC request dictionary
            
        Returns:
            List of nodes representing the complete routing path
        """
        # Build ordered list of waypoints: source → VNFs → destination
        waypoints = [request['source']]
        for vnf in request['chain']:
            if vnf['id'] in placement:
                waypoints.append(placement[vnf['id']])
        waypoints.append(request['destination'])
        
        # Compute path segment by segment using modified Dijkstra
        full_path = []
        security_policy = request['security_policy']
        
        for i in range(len(waypoints) - 1):
            src, dst = waypoints[i], waypoints[i+1]
            
            # Custom weight function for Dijkstra
            def weight(u, v, d):
                return self._compute_link_weight(u, v, security_policy)
            
            try:
                segment = nx.shortest_path(self.G, src, dst, weight=weight)
                # Add segment to full path (avoid duplicate waypoints)
                full_path.extend(segment[:-1])
            except nx.NetworkXNoPath:
                logger.error(f"No secure path from {src} to {dst}")
                raise RuntimeError(f"Routing failed: no path satisfying {security_policy} policy")
        
        # Add final destination
        full_path.append(waypoints[-1])
        
        return full_path
    
    def _compute_path_latency(self, path: List[str], placement: Dict, 
                            request: Dict) -> float:
        """
        Compute total end-to-end latency for a given path and placement.
        
        Components:
          - Propagation delay: sum of link latencies
          - Processing delay: sum of VNF processing times
          - Queuing delay: simplified exponential model
        """
        # Propagation + transmission delay along path
        link_latency = sum(
            self.G.edges[path[i], path[i+1]].get('latency', 1.0)
            for i in range(len(path) - 1)
        )
        
        # Processing delay for each VNF in chain
        proc_latency = sum(
            vnf['processing_time'] for vnf in request['chain']
        )
        
        # Queuing delay (simplified M/M/1 model approximation)
        # Scale with traffic load and node utilization
        avg_utilization = np.mean([
            self.resource_usage[node]['cpu'] / self.G.nodes[node]['cpu_capacity']
            for node in set(path)
        ]) if path else 0
        queue_latency = np.random.exponential(scale=0.5 * (1 + avg_utilization))
        
        total_latency = link_latency + proc_latency + queue_latency
        return total_latency
    
    def _check_security_compliance(self, path: List[str], 
                                 request: Dict) -> bool:
        """
        Verify that entire path satisfies security policy requirements.
        """
        min_policy = request['security_policy']
        
        for i in range(len(path) - 1):
            edge = (path[i], path[i+1])
            edge_security = self.G.edges[edge].get('security_level', 'none')
            
            if not self._meets_policy(edge_security, min_policy):
                logger.debug(f"Edge {edge} security '{edge_security}' < required '{min_policy}'")
                return False
        
        return True
    
    def _meets_policy(self, edge_security: str, policy: str) -> bool:
        """Check if edge security level meets policy requirement."""
        security_levels = {'none': 0, 'basic': 1, 'advanced': 2}
        policy_requirements = {'low': 0, 'medium': 1, 'high': 2}
        
        edge_level = security_levels.get(edge_security, 0)
        required_level = policy_requirements.get(policy, 0)
        
        return edge_level >= required_level
    
    def process_request(self, request: Dict) -> Dict:
        """
        Process a single SFC request through both algorithm phases.
        
        Args:
            request: SFC request dictionary
            
        Returns:
            Dictionary with placement, path, latency, and compliance results
        """
        request_id = request['id']
        logger.debug(f"Processing request {request_id}: {request['security_policy']} security, {request['max_latency']}ms max latency")
        
        try:
            # Phase 1: Place VNFs
            placement = self.place_vnfs(request)
            
            # Phase 2: Route path through placed VNFs
            path = self.route_path(placement, request)
            
            # Compute end-to-end latency
            total_latency = self._compute_path_latency(path, placement, request)
            
            # Check security compliance
            compliant = self._check_security_compliance(path, request)
            
            # Determine if request is accepted (meets latency bound)
            accepted = total_latency <= request['max_latency']
            
            result = {
                'request_id': request_id,
                'placement': placement,
                'path': path,
                'latency_ms': total_latency,
                'compliant': compliant,
                'accepted': accepted,
                'error': None
            }
            
            logger.debug(f"Request {request_id}: latency={total_latency:.2f}ms, accepted={accepted}, compliant={compliant}")
            return result
            
        except Exception as e:
            logger.error(f"Failed to process request {request_id}: {str(e)}")
            return {
                'request_id': request_id,
                'placement': {},
                'path': [],
                'latency_ms': float('inf'),
                'compliant': False,
                'accepted': False,
                'error': str(e)
            }
